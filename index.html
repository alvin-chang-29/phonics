// 控制彈窗開關 function toggleBatchModal() { const modal = document.getElementById('batch-modal'); modal.classList.toggle('hidden'); } // 核心批次處理邏輯 async function processBatch() { const fileInput = document.getElementById('file-input'); const textArea = document.getElementById('batch-text'); let rawContent = ""; // 1. 優先處理檔案 if (fileInput.files.length > 0) { const file = fileInput.files[0]; rawContent = await file.text(); } // 2. 若沒檔案則處理 Textarea else if (textArea.value.trim() !== "") { rawContent = textArea.value; } else { alert("請選擇檔案或輸入文字內容！"); return; } // 3. 解析內容 (支援換行或逗號區隔) // 使用正規表達式 [,\n\r]+ 分割，並過濾掉空白項 const rawList = rawContent.split(/[,\n\r]+/).map(item => item.trim()).filter(item => item !== ""); // 4. 迭代加入系統 rawList.forEach(text => { const lowerText = text.toLowerCase(); const newItem = { id: Date.now() + Math.random(), // 避免批次加入時 ID 重複 text: lowerText, category: autoCategorize(lowerText), // 自動歸類邏輯 (LongA, LongE, ShortA) type: currentMode // 根據當前分頁歸類 (單字或例句) }; items.push(newItem); }); // 5. 儲存並更新介面 localStorage.setItem('phonics_data', JSON.stringify(items)); renderList(); // 6. 清空並關閉 fileInput.value = ""; textArea.value = ""; toggleBatchModal(); alert(`成功匯入 ${rawList.length} 個項目！`); }
